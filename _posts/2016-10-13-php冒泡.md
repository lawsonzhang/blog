	$arr = [201, 3, -123, 261, 78, 35, 799, 1024, 1112, 235, 136, 138, 249, 1245, -25, 7, 889, 9951, 78, -1, 85, 0, 783, 2014, 93.2, -3.2];

	/**
	 * 冒泡排序
	 *
	 * 循环使用相邻两个元素的值进行比较
	 * 需要中间变量 用于调换位置 大的往后，小的往前
	 */
	function maopao($arr)
	{
		(string) $tem = '';
		for ($i=0; $i < count($arr)-1; $i++) { 
			for ($j=0; $j < count($arr)-1-$i; $j++) { 
				if ($arr[$j] > $arr[$j+1]) {
					$tem = $arr[$j];
					$arr[$j] = $arr[$j+1];
					$arr[$j+1] = $tem;
				}
			}
		}
		return $arr;
	}
